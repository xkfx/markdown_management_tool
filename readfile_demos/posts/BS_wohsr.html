6.51

测试代码：

#include <iostream>
using namespace std;
void f()
{
    cout << "f1" << endl;
}
void f(int)
{
    cout << "f2" << endl;
}
void f(int, int)
{
    cout << "f3" << endl;
}
void f(double, double = 3.14)
{
    cout << "f4" << endl;
}
int main()
{
    // f(2.56, 42);  error: call of overloaded 'f(double, int)' is ambiguous
    f(42);
    f(42, 0);
    f(2.56, 3.14);
    return 0;
}

输出结果：

$ a
f2
f3
f4
