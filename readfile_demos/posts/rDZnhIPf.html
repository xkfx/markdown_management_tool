    echo -e "=======end===============\n"
}

list="we:are:champion"
ifs_old=$IFS
IFS=$':'
for val in $list
do
  echo "n=$val"
done
IFS=$ifs_old; endl

ifs_old=$IFS
IFS=''
echo $@,$*

#!/bin/bash
for file in `ls | grep .txt`
do
    NEW_FILE=`echo $file`
    # ${string/substring/replacement} # substring可以是正则表达式
    NEW_FILE=${NEW_FILE/.txt/.doc}
    mv $file $NEW_FILE
done
