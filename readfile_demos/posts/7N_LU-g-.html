        p1.say();
        p2.say();
    }

 */
public abstract class Person {
    abstract void say();
}
public class Chinese extends Person {
    public void say() {
        System.out.println("你好");
    }
}
public class American extends Person {
    public void say() {
        System.out.println("Hello");
    }
}
