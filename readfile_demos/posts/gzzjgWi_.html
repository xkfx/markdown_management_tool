#include <iostream>
#include <string>
using namespace std;
bool capital_begin(string word) {
    if (word[0] >= 'A' && word[0] <= 'Z') {
        return true;
    }
    return false;
}
int main()
{
    /*
            输入一组字符串，例如a b c d e e输出为“没有以大写字母开头的单词重复出现~”
        若输入为a b c d e F F j k则输出为F，这样的话增加一个条件就好了。
     */
    string word, lastw;
    cin >> lastw; // 初始化
    bool has_sameword = false;
    bool lastW = false;
    while (cin >> word) {
        if (word == lastw && capital_begin(word)) {
            cout << word << endl;
            has_sameword = true;
            break;
        } else {
            lastw = word;
        }
    }
    if (!has_sameword) {
        cout << "没有以大写字母开头的单词重复出现~" << endl;
    }
    return 0;
}
