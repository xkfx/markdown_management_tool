#include <iostream>

// -------------model------------------
// ---object---
class Student {

friend ostream &print(ostream&, const Student&);
friend istream &read(istream&, Student&);

public:
    Student() {}
    Student(string name, int sex, int birth, string home, string major, double score):
        name(name), sex(sex), birth(birth), home(home), major(major), score(score) {}

    void setName(const string& name) { this->name = name; }
    string getName() const { return name; }

    void setSex(int sex) { this->sex = sex; }
    int getSex() const { return sex; }

    void setBirth(int birth) { this->birth = birth; }
    int getBirth() const { return birth; }

    void setHome(const string& home) { this->home = home; }
    string getHome() const { return home; }

    void setMajor(const string& major) { this->major = major; }
    string getMajor() const { return major; }

    void setScore(double score) { this->score = score; }
    double getScore() const { return score; }

private:
    // basic info
    string name;
    int sex;
    int birth;
    string home;

    // details
    string major;    
    double score;
};

// print
ostream &print(ostream &os, const Student &person)
{
    os << person.name << " " << person.sex << " " << person.birth << " " << person.home
        << " " << person.major << " " << person.score;
    return os;
}

// read
istream &read(istream &is, Student &person)
{
    is >> person.name >> person.sex >> person.birth >> person.home >> person.major >> person.score;
    return is;
}


// ---fileIO----
vector<Student> student;
string FILENAME;

void setIO(const string& str) {
    FILENAME = str;
}

void readFile() {
    ifstream input(FILENAME.c_str());
    Student x;
    while (read(input, x)) {
        student.push_back(x);
    }
}

void writeFile() {
    ofstream output(FILENAME.c_str());
    for (int i = 0; i != student.size(); ++i) {
        print(output, student[i]) << endl;
    }
}

// ---CRUD------
void insertStudent(Student &person) {
    student.push_back(person);
}

void removeStudent(const string& name) {
    for (vector<Student>::iterator it = student.begin();
     it != student.end(); ++it) {  
        if (name == it->getName()) { 
            student.erase(it);
            break;
        }
    }  
}

Student& getStudent(const string& name) {
    for (vector<Student>::iterator it = student.begin();
     it != student.end(); ++it) {  
        if (name == it->getName()) { 
            return *it;
        }
    }  
}

// -----functions--
void importInfo(const string& file) {
    setIO(file.c_str());
    readFile();
    setIO(FILENAME.c_str());
}

void listStudentByMajor(const string& major) {
    for (vector<Student>::iterator it = student.begin();
     it != student.end(); ++it) {  
        if (major == it->getMajor()) { 
            print(cout, *it) << endl;
        }
    }  
}

void listStudentBySex(int sex) {
    for (vector<Student>::iterator it = student.begin();
     it != student.end(); ++it) {  
        if (sex == it->getSex()) { 
            print(cout, *it) << endl;
        }
    }  
}

void listStudentByAge(int age) {
    for (vector<Student>::iterator it = student.begin();
     it != student.end(); ++it) {  
        if (age == NOW-it->getBirth()/10000) { 
            print(cout, *it) << endl;
        }
    }  
}
