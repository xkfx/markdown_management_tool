1、单词计数

#include<stdio.h>

#define IN 1
#define OUT 0

main()
{
    int c, state;
    int nword;

    nword = 0;
    state = OUT;
    while((c = getchar()) != EOF){
        if(c == ' ' || c == '\n' || c == '\t')
            state = OUT;
        else if(state == OUT){
            state = IN;
            ++ nword;
        }
    }    // 可以用集合里的Vn图理解, 每次循环都有三种情况。
    printf("%d\n", nword);
}

int getline(char s[], int lim)
{
    int c, i;
    
    i = 0;
    while(--lim > 0 && (c=getchar()) != EOF && c != '\n')
        s[i++] = c;
    if(c == '\n')
        s[i++] = c;
    s[i] = '\0';
    return i;
}

int strindex(char s[], char t[])
{
    int i, j, k;
    
    for(i = 0; s[i] != '\0'; i ++){
        for(j=i, k=0; t[k]!='\0' && s[j]==t[k]; j++, k++)
            ;
        if(k > 0 && t[k] == '\0')
            return i;
    }
    return -1;
}
