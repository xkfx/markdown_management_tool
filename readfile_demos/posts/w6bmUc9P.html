1、用filter求素数。

埃拉托色尼筛选法。

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
def _odd_iter():
    n = 1
    while True:
        n = n + 2
        yield n
# 这是一个从3开始的无限奇数序列

def _not_divisible(n):
    return lambda x: x % n > 0
# 动态的筛选函数，如果不能被n整除就返回True

def primes():
    yield 2
    it = _odd_iter()
    while True:
        n = next(it)
        yield n
        it = filter(_not_divisible(n), it) #不是n的倍数的留下！

for n in primes():
    if n < 1000:
        print(n)
    else:
        break
