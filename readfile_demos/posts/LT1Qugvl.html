#include<stdio.h>
#include<string.h>
#define N 10
void sortString(char str[][50])
{
    char temp[];

    for(int i = 0; i < N; i ++)
        for(int j = 0; j < N-i-1; j ++)
            if(strcmp(str[j] , str[j+1]) < 0){
                strcpy(temp , str[j]);    
                strcpy(str[j] , str[j+1]);
                strcpy(str[j] , temp);
            }
}
