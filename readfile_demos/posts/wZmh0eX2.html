#include<stdio.h>
#include<stdlib.h>

void main(void)
{
    // locate
    char infile[20], outfile[20];
    gets(infile);
    gets(outfile);

    // open
    FILE *in, *out;
    in = fopen(infile, "r");
    if(in == NULL)
        printf("error: without infile\n");
    else
        printf("infile has open.\n");
    out = fopen(outfile, "w");

    // operate
    while(!feof(in))
        fputc(fgetc(in), out);

    // close
    if(fclose(in) == 0 && fclose(out) == 0)
        printf("close successfully\n");
    else
        printf("fclose: error\n");
    return;
}
