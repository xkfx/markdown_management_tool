#include<stdio.h>

void print_string(char s[])
{
    printf("%s", s);
}

int strlen(char *s)
{
    int n;

    for(n = 0; *s != '\0'; s ++)
        n ++;
    return n;
}

int main()
{
    char s[] = "hello world!";

    for(int i = 0; i <= strlen(s); i ++){
        print_string(s + i);
        printf("\n");
    }
}
