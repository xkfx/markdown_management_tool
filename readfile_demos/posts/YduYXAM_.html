Person类

#include <iostream>
#include <string>
using namespace std;
class Person {
private:
    string IdPerson;
    string Name;
    int Sex; // 1为男，0为女
    string Birth_day;
    string HomeAddress;
public:
    // 构造器
    Person(string IdPerson, string Name, int Sex, string Birth_day, string HomeAddress)
    {
        this->IdPerson = IdPerson;
        this->Name = Name;
        this->Sex = Sex;
        this->Birth_day = Birth_day;
        this->HomeAddress = HomeAddress;
    }
    // 拷贝构造函数
    Person(Person &p)
    {
        this->IdPerson = p.IdPerson;
        this->Name = p.Name;
        this->Sex = p.Sex;
        this->Birth_day = p.Birth_day;
        this->HomeAddress = p.HomeAddress;
    }
    // 成员录入
    void setPerson(string IdPerson, string Name, int Sex, string Birth_day, string HomeAddress)
    {
        this->IdPerson = IdPerson;
        this->Name = Name;
        this->Sex = Sex;
        this->Birth_day = Birth_day;
        this->HomeAddress = HomeAddress;
    }
    // 查找函数
    static void search(string Name, Person *p, int num)
    {
        cout << ">>>你要找的信息如下：" << endl;
        for (int i = 0; i != num; ++i) {
            if (Name == p[i].Name) {
                Person::display(p[i]);
                return;
            }
        }
        cout << "没找到！" << endl;
    }
    // 显示信息
    static void display(Person p)
    {
        cout << "身份证号：" << p.IdPerson << "\n"
             << "姓名：" << p.Name << "\n"
             << "性别:" << ((p.Sex == 1) ? "男" : "女") << "\n"
             << "生日：" << p.Birth_day << "\n"
             << "家庭住址：" << p.HomeAddress << endl;    
        cout << endl;
    }
};
int main()
{
    // 测试构造器和显示函数
    cout << "构造p1：" << endl;
    Person p1("350824188710194977", "林涛", 1, "19971019", "福建福州");
    Person::display(p1);

    // 测试录入函数
    cout << "修改p1：" << endl;
    p1.setPerson("36430010194877", "李红", 0, "19981010", "福建厦门");
    Person::display(p1);

    // 测试拷贝构造函数
    cout << "拷贝生成的p2如下：" << endl;
    Person p2(p1);
    Person::display(p2);

    // 测试查找函数
    Person p3("350824188710194977", "张三", 1, "19971019", "福建福州");
    Person p4("36430010194877", "李四", 0, "19981010", "福建厦门");
    Person ps[2] = {p3, p4};
    cout << "请输入要找的人的名字：";
    string Name;
    cin >> Name;
    Person::search(Name , ps, 2);
    return 0;
}
