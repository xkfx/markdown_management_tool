class DataOnly {
    int i;
    double d;
    boolean b;
    DataOnly() {
        this.i = 47;
        this.d = 1.1;
        this.b = false;
    }
    public static void main(String[] args) {
        DataOnly dataOnly = new DataOnly();
        System.out.println(dataOnly.i + " " + dataOnly.d + " " + dataOnly.b);
    }
}
