char *Random(char* word) {
    int len = rand() % 99;
    for (int i = 0; i != len; ++i) 
        word[i] = ch[rand()%26 + 1];
    word[len] = '\0';
    return word;
}

void make() {
    srand(unsigned(time(0)));
    char word[100];
    for (int i = 0; i != 10000; ++i) {
        cout << Random(word) << endl;         
    }
}

int main() {
    setOut("Random");
    make();
    return 0;
}
