int main()
{

    /*
     * 
     * 【1-多行注释】
     */
     
    // 输出 hello world <--- 【2-单行注释】
    std::cout << "Hello, World" << std::endl;
    int v1 = 0, v2 = 0; // 声明两个变量 <---【3-半行注释】
    std::cin >> v1 >> v2;
    return 0;
}

#include<iostream>
int main()
{
    int sum = 0, val = 50;
    while(val <= 100){
        sum += val;
        ++val;
    }
    std::cout << "Sum of 50 to 100 inclusive is " << sum << std::endl;
    return 0;
}

练习1.11

#include<iostream>
int main()
{
    std::cout << "Enter two numbers: " << std::endl;
    int v1 = 0, v2 = 0;
    std::cin >> v1 >> v2;
    while(v1 <= v2){
        std::cout << v1++ << std::endl;
    }
    return 0;
}

#include<iostream>
int main()
{
    int sum = 0, value = 0;
    
    while(std::cin >> value)    // 命令行下用 ctrl + z 结束输入， 或是键入一个无效输入
        sum += value;
    std::cout << "Sum is: " << sum << std::endl;
    return 0;
}
