链接法解决冲突

<!DOCTYPE html>
<html>
    <head>
        <meta charset="UTF-8">
        <title></title>
    </head>
    <body>
        <script type="text/javascript">
            let makeObj = function(key, value, n=null, p=null) {
                let obj = {
                    key: key, 
                    value: value,
                    next: n,
                    prev: p,
                };
                return obj;
            };            
            /**
             * 散列表的每一个槽都对应一个双向循环列表。
             */
            function makeLinkedList(arr) {
                let list = {};
                list.nil = makeObj(null, null);
                let temp = list.nil; 
                for (let i = 0; i != arr.length; ++i) { 
                    temp.next = makeObj(arr[i].key, arr[i].data);
                    temp.next.prev = temp;
                    temp = temp.next;
                }
                list.nil.prev = temp; 
                temp.next = list.nil; 
                return list;
            }    
            
            function listDelete(list, obj) {
                // 要拿到相应元素的引用，并且确保obj确实在list里
                obj.prev.next = obj.next;
                obj.next.prev = obj.prev;                
            }    
            
            function listSearch(list, key) {
                let temp = list.nil.next;
                while (temp != list.nil && temp.key != key) {
                    temp = temp.next;
                }
                return temp;
            }

            function listInsert(list, obj) {
                obj.next = list.nil.next;
                list.nil.next.prev = obj;
                list.nil.next = obj;
                obj.prev = list.nil;
            }
        </script>
        
        <script>
            /*
            直接寻址表的缺点非常明显：
            例如说全域U是R，而实际只存储几个数字
            那么就非常浪费空间。（虽然最坏查找时间是O(1)）
            而散列表的优势在于：不仅表中查找一个元
            素的平均时间是O(1)，还能够节省大量空间
            但是存在一个问题：两个关键字可能映射到同一个槽中
            我们称这种情况为冲突，一方面可以通过
            精心设计散列函数减少冲突，但完全避免
            冲突是不可能的，所以另一方面我们必须要
            有解决冲突的办法，一种简单的方式就是链接法。
            */
            
            let keys = [0, 1, 2, 3]; // 关键字的全域U
            let chainedHashTable = [null, null, null, null]; 
            for (let i = 0; i != chainedHashTable.length; ++i) {
                chainedHashTable[i] = makeLinkedList([]);
            } // 表的每一个槽对应全域U中的一个关键字，以及一个双向链表↑
            
            chainedHashTable.search = function(k) {
                return listSearch(this[k], k);
            }
            
            chainedHashTable.insert = function(obj) {
                listInsert(this[obj.key], obj);
            }
            
            chainedHashTable.delete = function(obj) {
                listDelete(this[obj.key], obj);
            }
            
            let table = chainedHashTable;
            let obj1 = makeObj(keys[1], 2);
            let obj2 = makeObj(keys[1], 3); // 键值相同，冲突！
            table.insert(obj1);
            table.insert(obj2);
            console.log("table=", table); // 这里没有输出实际情况而是输出最终结果？
            console.log("obj=", table.search(obj1.key));
            table.delete(obj1);
            console.log("table=", table);
        </script>
    </body>
</html>

上面的代码忘记引入散列函数h(k)了。。最坏情况就是所有关键字散列到同一个槽中，从而产生一个长度为n的链表。它的平均性能实际依赖于所选取的散列函数h。

            // 开放寻址法的删除操作较复杂，在必须删除关键字的应用中，最好采用链接法解决冲突！
            
            let makeObj = function(key, value) {
                let obj = {
                    key: key, 
                    value: value,
                };
                return obj;
            };    
            
            let openAddressingTable = [null, null, null, null, null];
            let m = openAddressingTable.length; // 槽位总数
            
            openAddressingTable.insert = function(obj) {
                let i = 0, j = h(obj.key, i);
                while (i != m) {
                    j = h(obj.key, i); // js貌似没有repeat循环，统一起见进行了一次重复运算。
                    if (this[j] == null) {
                        this[j] = obj;
                        return j;
                    } else {
                        i = i + 1;
                    } // 检查槽位是否为空，不为空就根据散列函数找下一个槽位
                } // 当散列表逐渐填满时，每个槽位最终都可以被考虑。
                return "hash table overflow";
            }
            
            openAddressingTable.search = function(key) {
                let i = 0, j = h(key, i);
                while (i != m && this[j] != null) {
                    j = h(key, i);
                    if (this[j].key == key) {
                        return j;
                    }
                    i = i + 1;
                } // 当散列表逐渐填满时，每个槽位最终都可以被考虑。
                return null;                
            }
        </script>

将散列表的槽位比喻为带有序号的公厕坑位，并假设每个人都对坑位序号的选择有着特殊且独一无二的癖好（言下之意，每个人对坑位序号的癖好或多或少是不同的，没有两个人具有完全相同的癖好，我们可以把这些癖好抽象为一些数字例如22,33,1,9等，简称“癖好值”），我们选用一些字母来代表不同的人，例如sb1、sb2、sb3，然后它们各自的癖好值为sb1.key、sb2.key，sb3.key，比如sb1最热衷蹲序号为sb1.key的坑。

★直接寻址表：如果所有可能上公厕的人很少的话（例如是一个规模超小的社区的公厕），就可以采用这个简单而有效的技术。因为人数少嘛，我们就直接根据每个人的癖好给所有人都提供专属的坑位，这样任何人上厕所都不会遇到冲突，蹲坑啊、找人啊（当然你得知道那个人的癖好值）都超级方便。

★散列表：直接寻址型公厕的缺点是非常明显的，为每个人提供确切匹配癖好的专属坑位，这在大多都是过分奢侈的！例如说，大型购物广场就完全不可能采用直接寻址型公厕，因为癖好值的范围实在太大了，并且即便你具备广阔的土地、钱（恐怕是天文数字）为所有人提供专属坑位，实际也并不会有多少人同时蹲坑啊（实际蹲坑的人往往是一个size小且动态的集合），这样你用来建造公厕的大部分空间都将浪费掉。散列型公厕就是为了解决这样的问题，我们得找一个这样的癖好转换函数h(k)，这个癖好转换函数能够确保将所有可能蹲坑的人的癖好值转换为实际存在的坑位序号，“人多坑少”的情况下也只能这样将就地蹲了！在大多数场景下，坑位数量总是远远低于可能蹲坑的人数。不过这也引入了一个新问题：两个人的癖好值可能转换为同一个坑位序号。我们称这种情况叫冲突，就是：即便将就地去蹲坑，坑位也不一定是空的。那就只好继续将就咯，毕竟坑位有限的情况下，能拉到屎才是最关键的。

理想的解决方法是避免冲突或者让冲突次数最小化，我们可以试图选择一个合适的癖好转换函数做到这一点。简单来讲就是让每个h(k)尽可能随机。理想状态下，有m个坑位的话，每个想拉屎的人选任一坑位的可能性都为1/m，这种情况下发生冲突的概率是最小的。当然，实际中完全避免冲突是不可能的，比如厕所满了呢？这种经典场景估计大家都有碰到过。
