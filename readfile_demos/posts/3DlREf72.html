#include<iostream>
class Time {
    public:
        void set_time(void);
        void show_time(void);
    private:
        int hour;
        int minute;
        int sec;
};
Time t;
int main()
{
    t.set_time();
    t.show_time();
    return 0;
}
void Time::set_time()
{
    std::cin >> hour >> minute >> sec;
}
void Time::show_time()
{
    std::cout << hour << ":" << minute 
        << ":" << sec << std::endl;
}

矩阵相加

#include<stdio.h>
#define N 3
int c[N][N];
void sum(int a[N][N], int b[N][N], int n);
int main()
{
    // 初始化
    int a[N][N] = {0};
    int b[N][N] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    // 求和
    sum(a, b, N);
    // 输出
    for (int i = 0; i != 3; ++i) {
        for (int j = 0; j != 3; ++j) {
            printf("%d ", c[i][j]);
        }
        printf("\n");
    }
    return 0;
}

void sum(int a[N][N], int b[N][N], int n)
{
    for (int i = 0; i != n; ++i) {
        for (int j = 0; j != n; ++j) {
            c[i][j] = a[i][j] + b[i][j];
        }
    }
}
