#include<stdio.h>

void strcpy(char *s, char *t)
{
    while((*t++ = *s++) != '\0')
        ;
    *t = *s;
}

int main()
{
    char *s = "how are you?";
    char *t;
    
    strcpy(s , t);
    printf("%s\n", s);
    printf("%s\n", t);
    return 0;
}
