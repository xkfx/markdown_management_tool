<!DOCTYPE html>
<html>
    <head>
        <meta charset="UTF-8">
        <title></title>
    </head>

    <body>
        <textarea id="big-textarea" placeholder="paste your origin html text here..." rows="30" cols="150">
        </textarea>
        <button id="btn1">回车2空格</button>
        <button id="btn2">英文括号2中文括号</button>
        
        <script type="text/javascript">
            let button1 = document.querySelector('#btn1');
            button1.addEventListener('click', event => {
                let textArea = document.querySelector('#big-textarea');
                let inputTxt = textArea.value;
                textArea.value = 回车2空格(inputTxt);
            });
            let button2 = document.querySelector('#btn2');
            button2.addEventListener('click', event => {
                let textArea = document.querySelector('#big-textarea');
                let inputTxt = textArea.value;
                textArea.value = 英文括号2中文括号(inputTxt);
            });
        </script>
                
        <script type="text/javascript">                    
            function 回车2空格(inputTxt) {
                let modifiedTxt = inputTxt.replace(/\n/g, " ");
                return modifiedTxt;
            }
            function 英文括号2中文括号(inputTxt) {
                let modifiedTxt1 = inputTxt.replace(/\(/g, "（");
                let modifiedTxt2 = modifiedTxt1.replace(/\)/g, "）");
                return modifiedTxt2;
            }
        </script>
    </body>

</html>
